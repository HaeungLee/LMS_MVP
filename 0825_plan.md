-- 기술 스택 및 트랙 관리
CREATE TABLE learning_tracks (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL, -- 'python_basics', 'html_css', 'react', etc.
    display_name VARCHAR(100) NOT NULL, -- 'Python 기초', 'HTML & CSS', 'React'
    category VARCHAR(50) NOT NULL, -- 'foundation', 'development', 'specialization'
    prerequisite_tracks TEXT[], -- 선수 트랙들
    difficulty_level INTEGER DEFAULT 1, -- 1: 기초, 2: 중급, 3: 고급
    estimated_hours INTEGER DEFAULT 20,
    description TEXT,
    created_at TIMESTAMP DEFAULT NOW()
);

-- 확장된 문제 테이블
ALTER TABLE questions ADD COLUMN track_id INTEGER REFERENCES learning_tracks(id);
ALTER TABLE questions ADD COLUMN sub_topic VARCHAR(100); -- 'variables', 'functions', 'css-grid', etc.
ALTER TABLE questions ADD COLUMN practical_focus BOOLEAN DEFAULT false; -- 실무 연계 여부
ALTER TABLE questions ADD COLUMN project_based BOOLEAN DEFAULT false; -- 프로젝트 기반 여부

-- 학습자 진도 추적 (트랙별)
CREATE TABLE user_track_progress (
    id SERIAL PRIMARY KEY,
    user_id INTEGER REFERENCES users(id),
    track_id INTEGER REFERENCES learning_tracks(id),
    total_questions INTEGER DEFAULT 0,
    correct_answers INTEGER DEFAULT 0,
    mastery_percentage FLOAT DEFAULT 0.0,
    current_sub_topic VARCHAR(100),
    estimated_completion_date DATE,
    started_at TIMESTAMP DEFAULT NOW(),
    last_activity TIMESTAMP DEFAULT NOW(),
    UNIQUE(user_id, track_id)
);

-- AI 참고자료 시스템
CREATE TABLE learning_resources (
    id SERIAL PRIMARY KEY,
    track_id INTEGER REFERENCES learning_tracks(id),
    sub_topic VARCHAR(100),
    resource_type VARCHAR(50), -- 'documentation', 'tutorial', 'video', 'project'
    title VARCHAR(200) NOT NULL,
    url TEXT NOT NULL,
    description TEXT,
    difficulty_level INTEGER DEFAULT 1,
    created_at TIMESTAMP DEFAULT NOW()
);
```

class CareerPathRecommendationEngine:
    def __init__(self):
        self.career_paths = {
            'fullstack_developer': {
                'tracks': ['python_basics', 'html_css', 'javascript_basics', 'react_basics', 'fastapi_backend', 'database_design'],
                'description': '풀스택 웹 개발자'
            },
            'data_scientist': {
                'tracks': ['python_basics', 'python_intermediate', 'data_analysis', 'machine_learning', 'deep_learning'],
                'description': '데이터 사이언티스트'
            },
            'saas_entrepreneur': {
                'tracks': ['python_basics', 'html_css', 'react_basics', 'fastapi_backend', 'data_analysis', 'cloud_deployment'],
                'description': 'SaaS 창업자'
            }
        }
    
    async def recommend_learning_path(self, user_id: int, career_goal: str, current_skills: List[str]) -> Dict:
        # 1. 커리어 목표별 필수 트랙 조회
        required_tracks = self.career_paths.get(career_goal, {}).get('tracks', [])
        
        # 2. 현재 스킬 레벨 평가
        current_progress = await self.assess_current_skills(user_id, current_skills)
        
        # 3. 개인화된 학습 순서 생성
        learning_sequence = await self.generate_learning_sequence(required_tracks, current_progress)
        
        # 4. 예상 완료 시간 계산
        estimated_timeline = self.calculate_timeline(learning_sequence)
        
        return {
            'career_goal': career_goal,
            'learning_path': learning_sequence,
            'estimated_months': estimated_timeline,
            'next_milestone': learning_sequence[0] if learning_sequence else None
        }
```

class AdaptiveDifficultyEngine:
    async def adjust_question_difficulty(self, user_id: int, track_id: int) -> str:
        # 1. 최근 10문제 정답률 분석
        recent_performance = await self.get_recent_performance(user_id, track_id)
        
        # 2. 학습 속도 분석 (문제당 소요 시간)
        learning_speed = await self.analyze_learning_speed(user_id, track_id)
        
        # 3. 동적 난이도 조정
        if recent_performance > 0.8 and learning_speed < 60:  # 80% 이상 정답률, 1분 미만
            return 'hard'
        elif recent_performance > 0.6:
            return 'medium'
        else:
            return 'easy'
            
    async def generate_adaptive_questions(self, user_id: int, track_id: int, count: int = 5) -> List[Question]:
        # 개인 실력에 맞는 문제 난이도로 생성
        optimal_difficulty = await self.adjust_question_difficulty(user_id, track_id)
        return await self.ai_question_generator.generate_questions(track_id, optimal_difficulty, count)

class LearningResourceRecommender:
    async def recommend_resources_after_quiz(self, user_id: int, question_id: int, was_correct: bool) -> List[Dict]:
        question = await self.get_question(question_id)
        track = question.track
        sub_topic = question.sub_topic
        
        if was_correct:
            # 정답시: 심화 학습 자료 추천
            return await self.get_advanced_resources(track.name, sub_topic)
        else:
            # 오답시: 기초 복습 자료 추천
            user_weaknesses = await self.analyze_weakness_pattern(user_id, sub_topic)
            return await self.get_remedial_resources(track.name, sub_topic, user_weaknesses)
    
    async def get_contextual_feedback_with_resources(self, user_id: int, answer: str, question: Question) -> Dict:
        # 1. 기존 AI 피드백 생성
        feedback = await self.generate_ai_feedback(user_id, answer, question)
        
        # 2. 관련 학습 자료 추천
        resources = await self.recommend_learning_resources(question.track.name, question.sub_topic)
        
        # 3. 실무 프로젝트 제안
